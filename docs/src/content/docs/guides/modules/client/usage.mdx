<TabItem label="Get">
```ts copy wrap twoslash showLineNumbers
/** This must be a unique secret string, use env variables */
const AUTH_KEY = process.env['AUTH'] as string
/** This must be the url of your database, either the IP or the URL given by the host, try to add the protocol (http or https). Also avoid ending with /*/
const HOST = process.env['HOST'] as string
import { DatabaseClient } from "@vimn/webdb"

const Client = new DatabaseClient({
  auth: AUTH_KEY,
  host: HOST,
  port: 80
});
// ---cut---
// Get the data from the table 'main'
console.log(await Client.get({ table: 'main' }));
// Get the raw data from the table 'main'
console.log(await Client.get.raw({ table: 'main' }));
// Gets the value of the key 'foo' in the table 'main'
console.log(await Client.get({ table: 'main', key: 'foo' }));
```
</TabItem>

