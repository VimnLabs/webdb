import { Steps } from '@astrojs/starlight/components';

<Steps>
1. <h4 style="margin-bottom: 0"> Universal compatibility: </h4>
<p style="margin-top: 0">Works seamlessly with any environment and purpose, allowing you to use it easily and quickly. Even regardless of whether you use TypeScript or JavaScript.</p>
2. <h4 style="margin-bottom: 0"> Easy initialization: </h4>
<p style="margin-top: 0">Focusing on simplicity, it is easy to set up and use.</p>
3. <h4 style="margin-bottom: 0"> Optimized development time: </h4>
<p style="margin-top: 0">Having the database remotely allows a migration of machines in a simple way, plus as everything is stored via JSON facilitates manual editing.</p>
4. <h4 style="margin-bottom: 0"> Ease of maintenance: </h4>
<p style="margin-top: 0">With a remote database, the code is cleaner and easier to maintain, facilitating future upgrades and enhancements. Plus you can have multiple instances accessing the same database.</p>
</Steps>
